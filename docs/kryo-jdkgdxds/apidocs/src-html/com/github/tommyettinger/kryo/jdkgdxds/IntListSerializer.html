018package com.github.tommyettinger.kryo.jdkgdxds;
019
020import com.esotericsoftware.kryo.Kryo;
021import com.esotericsoftware.kryo.Serializer;
022import com.esotericsoftware.kryo.io.Input;
023import com.esotericsoftware.kryo.io.Output;
024import com.github.tommyettinger.ds.IntList;
025
026/**
027 * Kryo {@link Serializer} for jdkgdxds {@link IntList}s.
028 */
029public class IntListSerializer extends Serializer<IntList> {
030
031    public IntListSerializer() {
032        setAcceptsNull(false);
033    }
034
035    @Override
036    public void write(final Kryo kryo, final Output output, final IntList data) {
037        int length = data.size();
038        output.writeInt(length, true);
039        for (int i = 0; i < length; i++)
040            output.writeVarInt(data.get(i), false);
041    }
042
043    @Override
044    public IntList read(final Kryo kryo, final Input input, final Class<? extends IntList> dataClass) {
045        int length = input.readInt(true);
046        IntList data = new IntList(length);
047        for (int i = 0; i < length; i++)
048            data.add(input.readVarInt(false));
049        return data;
050    }
051
052    @Override
053    public IntList copy(Kryo kryo, IntList original) {
054        return new IntList(original);
055    }
056}
